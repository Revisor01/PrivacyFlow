4. **STAB-03 — Print-Audit gesamt:**
   ```bash
   python3 -c "
   import os
   unwrapped = 0
   for root, dirs, files in os.walk('.'):
       if '.build' in root or 'Pods' in root: continue
       for f in files:
           if f.endswith('.swift'):
               path = os.path.join(root, f)
               with open(path) as fh:
                   lines = fh.readlines()
               in_debug = False
               for i, line in enumerate(lines):
                   stripped = line.strip()
                   if stripped == '#if DEBUG': in_debug = True
                   elif stripped == '#endif': in_debug = False
                   elif 'print(' in stripped and not in_debug and not stripped.startswith('//'):
                       unwrapped += 1
                       print(f'  UNWRAPPED: {path}:{i+1}: {stripped}')
   print(f'Total unwrapped print() calls: {unwrapped}')
   "
   # Erwartung: 0 unwrapped
   ```
